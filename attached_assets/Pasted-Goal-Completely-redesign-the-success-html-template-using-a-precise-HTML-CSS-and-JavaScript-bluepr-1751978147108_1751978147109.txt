<script>
  function copyIdToClipboard() {
    const idText = document.getElementById('submission-id-text').innerText;
    navigator.clipboard.writeText(idText).then(() => {
      const copyBtn = document.getElementById('copy-btn');
      copyBtn.innerHTML = '<i class="fas fa-check"></i>';
      setTimeout(() => {
        copyBtn.innerHTML = '<i class="far fa-copy"></i>';
      }, 2000);
    });
  }

  function showLanguage(selectedLang) {
    // Hide all text elements
    document.querySelectorAll('.lang-text').forEach(el => el.classList.remove('active'));
    // Deactivate all tabs
    document.querySelectorAll('.lang-tab').forEach(el => el.classList.remove('active'));

    // Show the selected one
    document.getElementById(selectedLang).classList.add('active');
    // Activate the clicked tab
    event.currentTarget.classList.add('active');
  }
</script>
